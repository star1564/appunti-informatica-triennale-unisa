> [!quote] <font color="orange">Teorema di Weierstrass</font>
>Una funzione $f:[a,b]\to\mathbb{R}$  ammette **almeno** un [[049 Massimi e Minimi Relativi e Assoluti#Massimo Assoluto|massimo assoluto]] ed un [[049 Massimi e Minimi Relativi e Assoluti#Minimo Assoluto|minimo assoluto]] se e solo se:
>1. $f$ è una [[035 Funzione Continua|funzione continua]] 
>2. $[a,b]$ è un intervallo [[004 Intervalli#Intervallo Chiuso|chiuso]] e [[004 Intervalli#Intervalli Limitati|limitato]]
>$$\color{#CC241D}M=\max_{[a,b]} f,\quad m=\min_{[a,b]}f$$

![[Pasted image 20241003155934.png|500]]

Questi valori coincidono con i [[050 Teorema di Fermat#Punto Stazionario|punti stazionari]].


> [!warning] I punti di massimo e di minimo possono essere non unici
> Ad esempio la [[FT1 Funzione Seno|funzione seno]].

function extractNumberFromString(inputString) {
	const match = inputString.match(/^\d{3}/);
	
	if (match) {
		return match[0];
	} else {
		return null; // Return null if no match is found
	}
}

function startsWithNumber(inputString, targetNumber) {
  // Use a regular expression to check if the string starts with the target number
  const pattern = new RegExp(`^${targetNumber}`);
  return pattern.test(inputString);
}

function sort2Array(array){
	let res = []
	
	if (array[0] == undefined)
		res.push(array[1], array[1])
	else if (array[1] == undefined)
		res.push(array[0], array[0])
	else if (parseInt(extractNumberFromString(array[0])) > parseInt(extractNumberFromString(array[1])))
		res.push(array[1], array[0])
	else
		res.push(array[0], array[1])
	
	return res
}
