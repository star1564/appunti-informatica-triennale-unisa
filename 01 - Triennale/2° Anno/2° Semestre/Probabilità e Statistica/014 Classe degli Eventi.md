>Dato un [[011 Evento|Evento]] $E$ dello [[010 Spazio Campionario|Spazio Campionario]] $S$, la **Classe degli Eventi** $\mathcal{F}$ è una famiglia di sottoinsiemi di $S$ tale che
>1. $S\in \mathcal{F}$
>2. Se $E\in \mathcal{F}$ allora $\overline{E} \in \mathcal{F}$
>3. Se $E_1,E_2,\ldots\in \mathcal{F}$ allora $E_1\cup E_2\cup \ldots\in \mathcal{F}$
>4. $\emptyset\in\mathcal{F}$
>5. Se $E_1,E_2,\ldots\in \mathcal{F}$ allora $E_1\cap E_2\cap \ldots\in \mathcal{F}$

È una famiglia che contiene:
- L'intero Spazio Campionario con tutti i suoi eventi
- L'[[011 Evento#^660fb4|Evento Certo]]
- L'[[011 Evento#^da36b3|Evento Impossibile]]

Questa è [[Chiusura di un_opearzione|chiusa]] rispetto all'unione, intersezione e complementazione.

function extractNumberFromString(inputString) {
	const match = inputString.match(/^\d{3}/);
	
	if (match) {
		return match[0];
	} else {
		return null; // Return null if no match is found
	}
}

function startsWithNumber(inputString, targetNumber) {
  // Use a regular expression to check if the string starts with the target number
  const pattern = new RegExp(`^${targetNumber}`);
  return pattern.test(inputString);
}

function sort2Array(array){
	let res = []
	
	if (array[0] == undefined)
		res.push(array[1], array[1])
	else if (array[1] == undefined)
		res.push(array[0], array[0])
	else if (parseInt(extractNumberFromString(array[0])) > parseInt(extractNumberFromString(array[1])))
		res.push(array[1], array[0])
	else
		res.push(array[0], array[1])
	
	return res
}
